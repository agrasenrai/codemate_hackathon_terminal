Available Commands

pwd                   # show current directory (Linux/macOS)
cd            # change directory
cd ..                 # go back one level
ls                    # list files (Linux/macOS)
ls -l                 # detailed list (Linux/macOS)
dir                   # list files (Windows)
mkdir           # make directory
md              # make directory (Windows alternative)
touch           # create empty file
echo 'text' > file    # write text to file
rm              # delete file (Linux/macOS)
del             # delete file (Windows)
rm -r         # delete folder (Linux/macOS)
rmdir         # delete folder (Windows)
cp a.txt b.txt        # copy file
copy a.txt b.txt      # copy file (Windows alternative)
mv a.txt b.txt        # move/rename (Linux/macOS)
move a.txt b.txt      # move/rename (Windows alternative)
cat file.txt          # show file contents (Linux/macOS)
type file.txt         # show file contents (Windows)
head file.txt         # show first 10 lines
tail file.txt         # show last 10 lines
wc file.txt           # count lines/words/chars
grep pattern file.txt # search for pattern in file
find . -name pattern  # search files by name
tree                  # show directory tree
nano file.txt         # edit file (Linux/macOS)
open file.txt         # open file (macOS)
open .                # open Finder (macOS)

whoami                # current user
hostname              # computer name (now always works)
uname -a              # system details (Linux/macOS, now always works)
sw_vers               # macOS version
systeminfo            # Windows system details
df -h                 # disk usage (Linux/macOS)
free -h               # memory usage (Linux/macOS)
echo $PATH            # show PATH variable (Linux/macOS)
echo %PATH%           # show PATH variable (Windows)
date                  # show current date/time
clear                 # clear terminal

ps aux                # list processes (Linux/macOS)
tasklist              # list processes (Windows)
top                   # live process monitor (Linux/macOS)
htop                  # better process monitor (if installed, Linux/macOS)
kill             # kill process (Linux/macOS)
taskkill /PID  /F # kill process (Windows)

ping google.com       # test connection
ifconfig              # show IP info (Linux/macOS)
ip a                  # modern Linux IP info
ipconfig              # show IP info (Windows)
netstat -an           # network connections (all OS)
netstat -tulpn        # open ports (Linux)
ss -tulpn             # modern Linux open ports
Get-NetTCPConnection  # PowerShell: list open ports (Windows)
lsof -i -P -n | grep LISTEN  # open ports (Linux/macOS)
